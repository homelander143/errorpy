def count_frequencies(numbers):
    """
    Create a dictionary representing the frequency of each number in the list.
    """
    frequency_dict = {}
    for number in numbers:
        if number in frequency_dict:
            frequency_dict[number] += 1
        else:
            frequency_dict[number] = 1
    return frequency_dict

def main():
    # Input the list of integers
    input_str = input("Enter a list of integers separated by spaces: ")
    
    # Convert the input string into a list of integers
    numbers = list(map(int, input_str.split()))
    
    # Get the frequency dictionary
    frequency_dict = count_frequencies(numbers)
    
    # Display the frequency dictionary
    print("Frequency of each number:")
    for number, frequency in frequency_dict.items():
        print(f"{number}: {frequency}")

if __name__ == "__main__":
    main()

def mirror_characters(s):
    """
    Find mirror characters in a given string.
    """
    # Define mirror character pairs
    mirror_pairs = {
        'b': 'd', 'd': 'b',
        'p': 'q', 'q': 'p',
        'A': 'A', 'H': 'H', 'I': 'I', 'M': 'M', 'O': 'O', 'T': 'T', 'U': 'U', 'V': 'V', 'W': 'W', 'X': 'X', 'Y': 'Y'
    }

    mirror_chars = set()
    
    for char in s:
        if char in mirror_pairs:
            mirror_char = mirror_pairs[char]
            if mirror_char in s:
                mirror_chars.add(char)
                mirror_chars.add(mirror_char)
    
    return sorted(mirror_chars)

def main():
    # Input string
    s = input("Enter the string: ")

    # Find mirror characters
    result = mirror_characters(s)

    # Display result
    print("Mirror characters in the string are:", result)

if __name__ == "__main__":
    main()

import re

def validate_ifsc_code(ifsc_code):
    """
    Validate the IFSC code using a regular expression.
    """
    # Define the regular expression for validating IFSC codes
    pattern = r'^[A-Z]{4}0[A-Z0-9]{6}$'
    
    # Use re.match to check if the IFSC code matches the pattern
    if re.match(pattern, ifsc_code):
        return True
    else:
        return False

def main():
    # Input IFSC code
    ifsc_code = input("Enter the IFSC code: ")

    # Validate the IFSC code
    if validate_ifsc_code(ifsc_code):
        print("The IFSC code is valid.")
    else:
        print("The IFSC code is invalid.")

if __name__ == "__main__":
    main()
