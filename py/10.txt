def categorize_characters(s):
    """Categorize characters in the string and count them."""
    numeric_count = 0
    alphabetic_count = 0
    special_count = 0
    
    for char in s:
        if char.isdigit():
            numeric_count += 1
        elif char.isalpha():
            alphabetic_count += 1
        else:
            special_count += 1

    return numeric_count, alphabetic_count, special_count

def main():
    # Read input from the user
    user_input = input("Enter a string: ")
    
    # Categorize characters and get counts
    numeric_count, alphabetic_count, special_count = categorize_characters(user_input)
    
    # Print results
    print(f"Total numeric characters: {numeric_count}")
    print(f"Total alphabetic characters: {alphabetic_count}")
    print(f"Total special characters: {special_count}")

if __name__ == "__main__":
    main()

def replace_or_insert(main_string, substring, new_string):
    """Replace or insert a substring in the main string."""
    if substring in main_string:
        # Replace the substring with the new string
        result = main_string.replace(substring, new_string)
        print("Substring found and replaced.")
    else:
        # Insert the substring in the middle of the main string
        middle_index = len(main_string) // 2
        result = main_string[:middle_index] + substring + main_string[middle_index:]
        print("Substring not found. Inserted at the middle.")
    
    return result

def main():
    # Read input from the user
    main_string = input("Enter the main string: ")
    substring = input("Enter the substring to search for: ")
    new_string = input("Enter the new string to replace with (if substring is found): ")
    
    # Process the string
    result = replace_or_insert(main_string, substring, new_string)
    
    # Print the result
    print("Resulting string:", result)

if __name__ == "__main__":
    main()

def convert_word_to_number(word):
    """Convert a numeric word to its corresponding number."""
    # Dictionary mapping words to numbers
    num_words = {
        'zero': 0, 'one': 1, 'two': 2, 'three': 3, 'four': 4,
        'five': 5, 'six': 6, 'seven': 7, 'eight': 8, 'nine': 9,
        'ten': 10, 'eleven': 11, 'twelve': 12, 'thirteen': 13, 
        'fourteen': 14, 'fifteen': 15, 'sixteen': 16, 'seventeen': 17, 
        'eighteen': 18, 'nineteen': 19,
        'twenty': 20, 'thirty': 30, 'forty': 40, 'fifty': 50, 
        'sixty': 60, 'seventy': 70, 'eighty': 80, 'ninety': 90,
        'hundred': 100
    }
    
    # Split the input word into parts
    parts = word.lower().split()
    
    # Initialize the total number
    total = 0
    current = 0
    
    for part in parts:
        if part in num_words:
            # Handle 'hundred' specially
            if part == 'hundred':
                current *= num_words[part]
            else:
                current += num_words[part]
        else:
            # If the word is not recognized, return None
            return None
    
    total += current
    return total

def main():
    # Read input from the user
    numeric_word = input("Enter a numeric word: ")
    
    # Convert the word to number
    number = convert_word_to_number(numeric_word)
    
    # Print the result
    if number is not None:
        print(f"The numeric value is: {number}")
    else:
        print("The provided word is not a valid numeric word.")

if __name__ == "__main__":
    main()
