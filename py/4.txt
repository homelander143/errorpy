def print_pattern():
    # Define the pattern as a list of strings
    pattern = [
        "         *",
        "      *    *",
        "    *   *    *",
        "*    *    *     *"
    ]
    
    # Print each line of the pattern
    for line in pattern:
        print(line)

def is_ugly_number(n):
    if n <= 0:
        return False

    # Divide n by 2, 3, and 5 as long as it's divisible by them
    for factor in [2, 3, 5]:
        while n % factor == 0:
            n //= factor

    # If the remaining number is 1, it is an ugly number
    return n == 1

def main():
    try:
        # Read an integer input from the user
        number = int(input("Enter a positive integer: "))
        
        # Check if the number is an ugly number
        if is_ugly_number(number):
            print(f"{number} is an ugly number.")
        else:
            print(f"{number} is not an ugly number.")
    
    except ValueError:
        print("Invalid input. Please enter a valid positive integer.")

if __name__ == "__main__":
    main()

from datetime import datetime

# Print the current date
current_date = datetime.now()
print(f"Current date: {current_date.strftime('%Y-%m-%d')}")

# Read the birth year from the user
birth_year = int(input("Enter your birth year: "))

# Calculate the current year
current_year = current_date.year

# Calculate the age
age = current_year - birth_year

# Determine the age group
if age < 0:
    age_group = "Invalid age"
elif age < 13:
    age_group = "Child"
elif age < 20:
    age_group = "Teenager"
elif age < 35:
    age_group = "Young Adult"
elif age < 60:
    age_group = "Adult"
else:
    age_group = "Senior"

# Print the age and the age group
print(f"Your age: {age}")
print(f"You belong to the age group: {age_group}")
